 Its as easy as...
  • Step 1: Find the monitoring/api/ folder in your server.

  • Step 2: Use this template to create a new JS file with the name format: my-endpoint.js

    var instance = function() {}

var server = false;

instance.route = '/api/exampleEndpoint/:name';

instance.configure = function(config) {
  server = config.server;
}

instance.render = function(req, res) {
  var data = {};

  // read parameter from route
  var name = req.params.name || false;

  // form response
  var data = {
    suppliedName: name
  };

  // send response
  res.jsonp(data);
}

module.exports = instance;
  • Step 3: Set the instance.route property to name your route.
  • Step 4: Implement the instance.render function to return your custom data.
  • Step 5: Use the instance.configuration method to pull out server configuration values defined in your server.js file.
